  // Auto-generate title & description from image using Gemini Vision
  async function generateFromImage(file){
    const statusDiv = document.getElementById('aiGenStatus');
    statusDiv.innerText = 'AI বিশ্লেষণ করছে, দয়া করে অপেক্ষা করুন...';
    statusDiv.style.color = 'blue';
    const base64 = await new Promise((resolve) => {
      const reader = new FileReader();
      reader.onload = (e) => resolve(e.target.result.split(',')[1]);
      reader.readAsDataURL(file);
    });
    try {
      const response = await fetch(`https://generativelanguage.googleapis.com/v1beta/models/gemini-1.5-flash:generateContent?key=${GEMINI_API_KEY}`, {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({
          contents: [{
            parts: [
              { text: "You are an agricultural expert. Look at this product image and generate a short product title (in Bengali) and a detailed description (in Bengali) about its taste, usage, benefits, and ideal for which cooking. Format: TITLE: ...\nDESCRIPTION: ..." },
              { inline_data: { mime_type: file.type, data: base64 } }
            ]
          }]
        })
      });
      const data = await response.json();
      const text = data.candidates[0].content.parts[0].text;
      const titleMatch = text.match(/TITLE:\s*(.+)/i);
      const descMatch = text.match(/DESCRIPTION:\s*(.+)/i);
      let title = titleMatch ? titleMatch[1].trim() : "তাজা পণ্য";
      let desc = descMatch ? descMatch[1].trim() : "খেত থেকে সদ্য তোলা উচ্চ মানের পণ্য।";
      document.getElementById('prodName').value = title;
      document.getElementById('prodDesc').value = desc;
      statusDiv.innerText = '✅ নাম ও বিবরণ অটো জেনারেট সম্পন্ন!';
      statusDiv.style.color = 'green';
    } catch(e){
      console.error(e);
      statusDiv.innerText = 'AI জেনারেট ব্যর্থ, ম্যানুয়ালি দিন';
      statusDiv.style.color = 'red';
    }
  }
